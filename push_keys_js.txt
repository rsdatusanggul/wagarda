\n        // ================= API KEY MANAGEMENT =================\n        const btnCreateKey = document.getElementById('btn-create-key');\n        const descInput = document.getElementById('new-key-desc');\n        const apiKeysTbody = document.querySelector('#api-keys-table tbody');\n\n        async function fetchApiKeys() {\n            try {\n                const res = await fetch('/api/keys');\n                if (!res.ok) throw new Error('Failed to fetch API keys');\n                const keys = await res.json();\n                \n                apiKeysTbody.innerHTML = '';\n                keys.forEach(k => {\n                    const tr = document.createElement('tr');\n                    \n                    const tdDesc = document.createElement('td');\n                    tdDesc.textContent = k.description;\n                    \n                    const tdKey = document.createElement('td');\n                    tdKey.innerHTML = \`<code style=\"user-select: all; background: #eee; padding: 2px 5px;\">${k.key}</code>\`;\n                    \n                    const tdTime = document.createElement('td');\n                    tdTime.textContent = new Date(k.created_at).toLocaleString();\n                    \n                    const tdAction = document.createElement('td');\n                    const btnDel = document.createElement('button');\n                    btnDel.textContent = 'Revoke';\n                    btnDel.className = 'delete-btn';\n                    btnDel.onclick = () => deleteApiKey(k.id);\n                    tdAction.appendChild(btnDel);\n\n                    tr.append(tdDesc, tdKey, tdTime, tdAction);\n                    apiKeysTbody.appendChild(tr);\n                });\n            } catch (err) {\n                console.error(err);\n            }\n        }\n\n        async function createApiKey() {\n            const desc = descInput.value.trim();\n            if (!desc) return alert('Please enter a description');\n            \n            try {\n                const res = await fetch('/api/keys', {\n                    method: 'POST',\n                    headers: { 'Content-Type': 'application/json' },\n                    body: JSON.stringify({ description: desc })\n                });\n                const data = await res.json();\n                if (data.success) {\n                    descInput.value = '';\n                    fetchApiKeys();\n                    alert('New API Key Generated: ' + data.key + '\n\nPlease copy it now, it will be visible in the table below.');\n                }\n            } catch (err) {\n                console.error(err);\n                alert('Error creating API key');\n            }\n        }\n\n        async function deleteApiKey(id) {\n            if (!confirm('Are you sure you want to REVOKE this API Key? Applications using it will break immediately.')) return;\n            try {\n                const res = await fetch(\`/api/keys/${id}\`, { method: 'DELETE' });\n                const data = await res.json();\n                if (data.success) {\n                    fetchApiKeys();\n                }\n            } catch (err) {\n                console.error(err);\n                alert('Error deleting API key');\n            }\n        }\n\n        if(btnCreateKey) btnCreateKey.addEventListener('click', createApiKey);\n        if(apiKeysTbody) fetchApiKeys();
